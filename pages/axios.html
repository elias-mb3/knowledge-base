Utilizar Axios para consumo de API

Aprendendo a utilizar Axios para o consumo de API com HTTPS.

Publicado 21 Novembro 2023

Installation

Rode os seguintes comandos:

                                
    npm init
    npm install
    npm install axios

Add to Index.js

                                
    const axios = require('axios')

    const url = "https://api.adviceslip.com/advice"
                                    
    async function getAdivice() {
        axios.get(url)
        .then(res => {
            let advice = res.data.slip.advice;
            console.log(`E a frase do dia é: ${advice}!`)
        }
        )
        .catch(err =>{
            console.log(err)
        })
        }
        getAdivice()
